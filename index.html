// ПОЛНОСТЬЮ ИСПРАВЛЕННЫЙ И ОБНОВЛЕННЫЙ ИИ function aiTurn() { if (gamePhase !== 'battle' || !activeUnit) return; let ai = activeUnit; let enemies = units.filter(u => u.team === 'player' && u.isAlive); let allies = units.filter(u => u.team === 'enemy' && u.isAlive); if (enemies.length === 0) return; function getDistance(u1, u2) { return Math.max(Math.abs(u1.r - u2.r), Math.abs(u1.c - u2.c)); } // --- ЛОГИКА МЕЧНИКА ИИ --- if (ai.typeKey === 'swordsman') { if (ai.hp < ai.maxHp * 0.6 && ai.cooldowns.heal === 0) { executeHeal(ai, ai); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } if (ai.hp < ai.maxHp * 0.5 && ai.cooldowns.buff === 0) { executeReinforce(ai, ai); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } } // --- УМНАЯ ЛОГИКА ЛУЧНИКА ИИ (С ЛЕЧЕНИЕМ И СКИПОМ ХОДА) --- if (ai.typeKey === 'archer') { let closestEnemyDist = Math.min(...enemies.map(e => getDistance(ai, e))); let hasPoison = ai.dotPoison && ai.dotPoison.length > 0; let isLowHp = ai.hp < ai.maxHp * 0.7; // 1. УСЛОВИЕ ДЛЯ СКИПА ХОДА И ЛЕЧЕНИЯ (Пассивка "Живой"): // Если на лучнике Яд ИЛИ мало HP, и при этом враг НЕ стоит к нему вплотную (дистанция > 1) if ((hasPoison || isLowHp) && closestEnemyDist > 1) { logMessage(`${ai.name} (ИИ) решил полностью пропустить ход для лечения от яда/ран!`); // Скипаем и движение, и действие, чтобы гарантированно сработала пассивка "Живой" (+20% HP) ai.hasMoved = false; ai.hasActed = false; renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } // 2. Если включен таймер ожидания "Града стрел" — он НЕ может атаковать if (ai.buffs.noAttackDuration > 0) { logMessage(`${ai.name} (ИИ) готовит Град стрел и не может атаковать! (${ai.buffs.noAttackDuration} ходов остав.)`); // Если нет яда/нужды лечиться — отступает от врагов let validMoves = getValidMoveCells(ai); if (validMoves.length > 0) { let bestFleePos = validMoves[0]; let maxDist = -1; validMoves.forEach(pos => { let minDistToEnemy = Math.min(...enemies.map(e => getDistance(pos, e))); if (minDistToEnemy > maxDist) { maxDist = minDistToEnemy; bestFleePos = pos; } }); ai.r = bestFleePos.r; ai.c = bestFleePos.c; ai.hasMoved = true; } renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } // 3. Если Град Стрел готов к выпуску (5 выстрелов) if (ai.arrowRainShotsLeft > 0) { let inRangeEnemies = enemies.filter(e => getDistance(ai, e) <= getUnitEffectiveRange(ai)); if (inRangeEnemies.length > 0) { handleAttackAction(ai, inRangeEnemies[0]); return; } } // 4. Активация навыка «Град стрел», если откатился if (ai.cooldowns.arrowRain === 0 && !ai.buffs.noAttackDuration && ai.arrowRainShotsLeft === 0) { selectAbility('arrowRain'); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } } // --- ЛОГИКА МАГА ИИ --- if (ai.typeKey === 'mage') { let injuredAlly = allies.find(a => a.hp < a.maxHp * 0.7); if (injuredAlly) { executeAttack(ai, injuredAlly); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } if (ai.cooldowns.summonSwordsman === 0 || ai.cooldowns.summonArcher === 0) { let emptyCells = []; for (let dr = -1; dr <= 1; dr++) { for (let dc = -1; dc <= 1; dc++) { let nr = ai.r + dr, nc = ai.c + dc; if (nr >= 0 && nr < ROWS && nc >= 0 && nc < COLS && !getUnitAt(nr, nc) && !getObstacleAt(nr, nc)) { emptyCells.push({ r: nr, c: nc }); } } } if (emptyCells.length > 0) { let type = (ai.cooldowns.summonSwordsman === 0) ? 'swordsman' : 'archer'; let pos = emptyCells[0]; let summoned = createUnit(type, ai.team, pos.r, pos.c, units.length + 1, 5); units.push(summoned); ai.cooldowns[type === 'swordsman' ? 'summonSwordsman' : 'summonArcher'] = 10; ai.hasActed = true; logMessage(`${ai.name} (ИИ) призвал ${summoned.name}!`); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } } if (ai.cooldowns.storm === 0) { for (let r = 0; r <= ROWS - 3; r++) { for (let c = 0; c <= COLS - 3; c++) { let count = 0; for (let dr = 0; dr < 3; dr++) { for (let dc = 0; dc < 3; dc++) { let u = getUnitAt(r + dr, c + dc); if (u && u.team !== ai.team) count++; } } if (count >= 2) { executeStormArea(r, c); ai.cooldowns.storm = 10; ai.hasActed = true; logMessage(`${ai.name} (ИИ) обрушил Чародейскую бурю!`); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } } } } if (ai.cooldowns.poisonZone === 0) { for (let r = 0; r <= ROWS - 4; r++) { for (let c = 0; c <= COLS - 4; c++) { let count = 0; for (let dr = 0; dr < 4; dr++) { for (let dc = 0; dc < 4; dc++) { let u = getUnitAt(r + dr, c + dc); if (u && u.team !== ai.team) count++; } } if (count >= 2) { executePoisonArea(r, c); ai.cooldowns.poisonZone = 10; ai.hasActed = true; logMessage(`${ai.name} (ИИ) применил Ядовитое облако!`); renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } } } } let bestPos = { r: ai.r, c: ai.c }; let maxAlliesNearby = -1; let validMoves = getValidMoveCells(ai); validMoves.push({ r: ai.r, c: ai.c }); validMoves.forEach(pos => { let alliesCount = 0; allies.forEach(a => { if (a !== ai && getDistance(pos, a) <= 1) alliesCount++; }); if (alliesCount > maxAlliesNearby) { maxAlliesNearby = alliesCount; bestPos = pos; } }); ai.r = bestPos.r; ai.c = bestPos.c; ai.hasMoved = true; renderBoard(); setTimeout(endCurrentUnitTurn, 600); return; } // --- ДВИЖЕНИЕ И АТАКА (ОБЩАЯ ДЛЯ МЕЧНИКА И ЛУЧНИКА) --- if (ai.typeKey === 'swordsman' || ai.typeKey === 'archer') { let validMoves = getValidMoveCells(ai); validMoves.push({ r: ai.r, c: ai.c }); let bestMove = validMoves[0]; let minDistance = 999; validMoves.forEach(pos => { enemies.forEach(e => { let d = getDistance(pos, e); if (d < minDistance) { minDistance = d; bestMove = pos; } }); }); ai.r = bestMove.r; ai.c = bestMove.c; ai.hasMoved = true; // Проверка атаки после ходьбы (только если нет запрета на атаку) if (ai.buffs.noAttackDuration === 0) { let inRangeEnemies = enemies.filter(e => getDistance(ai, e) <= getUnitEffectiveRange(ai)); if (inRangeEnemies.length > 0) { handleAttackAction(ai, inRangeEnemies[0]); return; } } } renderBoard(); setTimeout(endCurrentUnitTurn, 600); }
